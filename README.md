-- To install, install docker, docker compose and follow the steps below --
1. download netkeeper and place in a directory on your docker host 
2. cd into your newly created directory with netkeeper files
3. run docker compose build --no-cache
4. run docker compose up -d
5. access app at http://ip-of-host:3001
6. admin/password = admin:admin
